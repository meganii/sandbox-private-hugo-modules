{{- /* Render Hook - 内部の.mdファイルリンクは適切なRelPermalinkに変換 - 外部リンクは適切な属性を付与 - アフィリエイトリンクは特別に処理 */ -}} {{- $isExternal := strings.HasPrefix .Destination "http" -}} {{- $isMdFile := strings.HasSuffix .Destination ".md" -}} {{- if and (not $isExternal) $isMdFile }} {{/* 内部マークダウンファイルへのリンク */}} {{- $url := urls.Parse .Destination -}} {{- $text := .Text | safeHTML -}} {{ with site.GetPage $url.Path }} {{ $text }} {{ end }} {{- else -}} {{/* 通常のリンク（内部または外部） */}} {{ .Text | safeHTML }} {{- end -}}
